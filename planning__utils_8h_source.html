97 void loadEigenVectorsFromParams() {
98 // Load the gravity vector
99 g_vec << 0, 0, -g;
100
101 // Load the reachability test points
102 reachability_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
103 -0.5 * robot_l, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
104 -0.5 * robot_w, 0, 0, 0, 0;
105 // Load the collision test points
106 collision_points_body << 0.5 * robot_l, -0.5 * robot_l, 0.5 * robot_l,
107 -0.5 * robot_l, 0, 0.5 * robot_w, 0.5 * robot_w, -0.5 * robot_w,
108 -0.5 * robot_w, 0, -0.5 * robot_h, -0.5 * robot_h, -0.5 * robot_h,
109 -0.5 * robot_h, -0.5 * robot_h;
110 }

187struct State {
188 Eigen::Vector3d pos;
189 Eigen::Vector3d vel;
190
191 bool operator==(const State rhs) const {
192 // Z velocity is overridden by action
193 return ((pos == rhs.pos) && (vel.head<2>() == rhs.vel.head<2>()));
194 }
195
196 bool operator!=(const State rhs) const {
197 // Z velocity is overridden by action
198 return ((pos != rhs.pos) || (vel.head<2>() != rhs.vel.head<2>()));
199 }
200
201 bool isApprox(const State rhs) const {
202 // Z velocity is overridden by action
203 return ((pos.isApprox(rhs.pos)) &&
204 (vel.head<2>().isApprox(rhs.vel.head<2>())));
205 }
206};
