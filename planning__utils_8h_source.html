100 // Load the gravity vector
101 G_VEC << 0, 0, -G_CONST;

103 // Load the reachability test points
104 reachability_points_body << 0.5 * ROBOT_L, -0.5 * ROBOT_L, 0.5 * ROBOT_L,
105 -0.5 * ROBOT_L, 0.5 * ROBOT_W, 0.5 * ROBOT_W, -0.5 * ROBOT_W,
106 -0.5 * ROBOT_W, 0, 0, 0, 0;
107 // Load the collision test points
108 collision_points_body << 0.5 * ROBOT_L, -0.5 * ROBOT_L, 0.5 * ROBOT_L,
109 -0.5 * ROBOT_L, 0, 0.5 * ROBOT_W, 0.5 * ROBOT_W, -0.5 * ROBOT_W,
110 -0.5 * ROBOT_W, 0, -0.5 * ROBOT_H, -0.5 * ROBOT_H, -0.5 * ROBOT_H,
111 -0.5 * ROBOT_H, -0.5 * ROBOT_H;
112 }

185// Define state with Eigen data
186struct StateEigen {
187 Eigen::Vector3d pos;
188 Eigen::Vector3d vel;
189
190 bool operator==(const StateEigen rhs) const {
191 // Z velocity is overridden by action
192 return ((pos == rhs.pos) && (vel.head<2>() == rhs.vel.head<2>()));
193 }
194
195 bool operator!=(const StateEigen rhs) const {
196 // Z velocity is overridden by action
197 return ((pos != rhs.pos) || (vel.head<2>() != rhs.vel.head<2>()));
198 }
199
200 bool isApprox(const StateEigen rhs) const {
201 // Z velocity is overridden by action
202 return ((pos.isApprox(rhs.pos)) &&
203 (vel.head<2>().isApprox(rhs.vel.head<2>())));
204 }
205};
