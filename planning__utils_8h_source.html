95// Define state with Eigen data
96struct StateEigen {
97 Eigen::Vector3d pos;
98 Eigen::Vector3d vel;
99
100 bool operator==(const StateEigen rhs) const {
101 // Z velocity is overridden by action
102 return ((pos == rhs.pos) && (vel.head<2>() == rhs.vel.head<2>()));
103 }
104
105 bool operator!=(const StateEigen rhs) const {
106 // Z velocity is overridden by action
107 return ((pos != rhs.pos) || (vel.head<2>() != rhs.vel.head<2>()));
108 }
109
110 bool isApprox(const StateEigen rhs) const {
111 // Z velocity is overridden by action
112 return ((pos.isApprox(rhs.pos)) &&
113 (vel.head<2>().isApprox(rhs.vel.head<2>())));
114 }
115};
